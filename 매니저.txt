    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getDatabase, ref, set, update, onValue, remove } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js";

        // =========================================================
        // [설정] 주문자와 똑같은 설정값을 여기에 붙여넣으세요.
        // =========================================================
        const firebaseConfig = {
            apiKey: "AIzaSyDn2Qoscan0DJsilwaoLzSMio7u_JrP9lI",
            authDomain: "order-901d1.firebaseapp.com",
            projectId: "order-901d1",
            storageBucket: "order-901d1.firebasestorage.app",
            messagingSenderId: "790529823560",
            appId: "1:790529823560:web:d70daa37b23a5b6912997d",
            measurementId: "G-2D27K14R4R"
        };
        // =========================================================

        const app = initializeApp(firebaseConfig);
        const db = getDatabase(app);

        // 전역 함수 연결
        window.changeShopStatus = changeShopStatus;
        window.updateStatus = updateStatus;
        window.resetAll = resetAll;
        window.stopAlert = stopAlert;

        document.getElementById('today-date-admin').innerText = new Date().toLocaleDateString('ko-KR');
        
        let lastOrderCount = 0;
        let isFirstLoad = true;

        // 1. 영업 상태 불러오기 & 실시간 감지
        onValue(ref(db, 'shopStatus'), (snapshot) => {
            const status = snapshot.val() || "OPEN";
            const radios = document.getElementsByName("shopStatus");
            for(let r of radios) {
                if(r.value === status) r.checked = true;
            }
        });

        // 2. 주문 내역 실시간 감지 (새 주문 오면 자동 실행됨)
        onValue(ref(db, 'orders'), (snapshot) => {
            const data = snapshot.val();
            const container = document.getElementById("order-container");
            
            if (!data) {
                container.innerHTML = "<div style='text-align:center; padding:50px; color:#888;'><h3>대기 중인 주문이 없습니다.</h3></div>";
                lastOrderCount = 0;
                return;
            }

            // 객체를 배열로 변환 (Firebase는 객체 형태임)
            const list = Object.values(data);
            
            // 알림 로직
            if (!isFirstLoad && list.length > lastOrderCount) {
                triggerAlert();
            }
            lastOrderCount = list.length;
            isFirstLoad = false;

            container.innerHTML = "";
            
            // 역순(최신순) 정렬 등 필요시 여기서 sort 가능. 여기선 등록순.
            list.forEach((order) => {
                const key = order.key; // 파이어베이스 키
                let statusClass = "";
                let actionAreaHtml = ""; 
                const requestMsg = order.request ? order.request : "요청사항 없음";

                if (order.status === "접수 대기") {
                    statusClass = "waiting";
                    actionAreaHtml = `
                        <span style="font-size:14px;">예상</span>
                        <input type="number" id="time-${key}" value="10">
                        <span style="font-size:14px; margin-right:5px;">분</span>
                        <button class="btn-accept" onclick="updateStatus('${key}', '조리 중'); event.stopPropagation();">접수</button>
                        <button class="btn-done" onclick="updateStatus('${key}', '조리 완료'); event.stopPropagation();">바로 완료</button>
                    `;
                } else if (order.status === "조리 중") {
                    statusClass = "cooking";
                    actionAreaHtml = `
                        <span class="scheduled-time">완료예정: ${order.completeTime}</span>
                        <button class="btn-gray">접수됨</button>
                        <button class="btn-done" onclick="updateStatus('${key}', '조리 완료'); event.stopPropagation();">조리 완료</button>
                    `;
                } else if (order.status === "조리 완료") {
                    statusClass = "done";
                    actionAreaHtml = `
                        <div style="background:#d4edda; color:#155724; padding:10px; text-align:center; font-weight:bold; border-radius:5px; width:100%;">
                            [완료] 픽업 대기중 (${order.completeTime})
                        </div>
                    `;
                }

                const html = `
                    <div class="order-card ${statusClass}">
                        <div class="top-row">
                            <span class="order-num">No. ${order.orderNumber}</span>
                            <span class="order-time">${order.timestamp} 접수</span>
                        </div>
                        <div class="user-info">
                            <span class="user-label">주문자:</span> ${order.name} | 
                            <span class="user-label">전화:</span> ${order.phone}
                        </div>
                        <span class="menu-text">${order.menu}</span>
                        <div class="request-box">
                            <span class="req-tag">[요청]</span> ${requestMsg}
                        </div>
                        <div class="btn-area">${actionAreaHtml}</div>
                    </div>
                `;
                container.innerHTML += html; // 최신순으로 하려면 prepend 사용
            });
        });

        // --- 기능 함수들 ---

        function triggerAlert() {
            playSirenOrder(3);
            document.body.classList.add("alert-mode");
        }

        function stopAlert() {
            document.body.classList.remove("alert-mode");
            window.speechSynthesis.cancel();
        }

        function playSirenOrder(count) {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel();
                const text = "주문이 들어왔습니다!";
                for(let i=0; i<count; i++) {
                    const msg = new SpeechSynthesisUtterance(text);
                    msg.lang = 'ko-KR';
                    msg.rate = 1.0; 
                    window.speechSynthesis.speak(msg);
                }
            }
        }

        function updateStatus(key, newStatus) {
            const updates = {};
            updates['/orders/' + key + '/status'] = newStatus;

            if (newStatus === '조리 중') {
                const minutes = parseInt(document.getElementById(`time-${key}`).value) || 10;
                const now = new Date();
                const finishTime = new Date(now.getTime() + minutes * 60000);
                updates['/orders/' + key + '/completeTime'] = finishTime.toLocaleTimeString('ko-KR', { hour: '2-digit', minute: '2-digit' });
            } else if (newStatus === '조리 완료') {
                updates['/orders/' + key + '/completeTime'] = new Date().toLocaleTimeString('ko-KR', { hour: '2-digit', minute: '2-digit' });
            }

            update(ref(db), updates);
            stopAlert();
        }

        function changeShopStatus(status) {
            set(ref(db, 'shopStatus'), status);
            alert("영업 상태가 공유되었습니다.");
        }

        function resetAll() {
            if (confirm("정말 마감하시겠습니까? 모든 주문 내역이 서버에서 삭제됩니다.")) {
                remove(ref(db, 'orders'));
                // 영업 상태도 초기화하려면 아래 주석 해제
                // set(ref(db, 'shopStatus'), 'OPEN');
                alert("초기화 완료");
            }
        }
